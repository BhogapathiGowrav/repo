FROM maven:3.9.6-eclipse-temurin-17 AS build

WORKDIR /app

COPY . .

RUN mvn clean package -DskipTests

FROM eclipse-temurin:17

WORKDIR /app

COPY --from=build /app/target/*.jar app.jar

ENTRYPOINT ["java", "-jar", "app.jar"]



pipeline {
 agent any

 tools {
   maven "Maven3"
 }

 environment {
   IMAGE_NAME = "calculator-app"
 }

 stages {

   stage('Checkout Code') {
     steps {
       git url: 'https://github.com/BhogapathiGowrav/calculator.git',
           branch: 'master'
     }
   }

   stage('Build Maven Project') {
     steps {
       bat 'mvn clean package -DskipTests'
     }
   }

   stage('Check Docker') {
     steps {
       bat 'docker --version'
     }
   }

   stage('Build Docker Image') {
     steps {
       bat 'docker build --no-cache -t calculator-app .'
     }
   }

   stage('Run Container') {
     steps {
       bat 'docker rm -f calculator-container || exit 0'
       bat 'docker run -d --name calculator-container -p 9090:8080 calculator-app'
     }
   }
 }

 post {
   success {
     echo 'Pipeline executed successfully!'
   }

   failure {
     echo 'Pipeline failed!'
   }
 }
}
